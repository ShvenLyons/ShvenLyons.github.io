<script>
(() => {
  const scene = document.getElementById('scene');
  const ship  = document.getElementById('ship');
  const targets = Array.from(document.querySelectorAll('.target'));
  const gotEl = document.getElementById('got');
  const done = document.getElementById('done');
  const again = document.getElementById('again');

  const bounds = () => scene.getBoundingClientRect();
  let w=0,h=0, x=100, y=100, vx=0, vy=0;
  let collected = [];

  function place(el){
    const pad=40;
    el.style.left = (Math.random()*(w-2*pad)+pad)+'px';
    el.style.top  = (Math.random()*(h-2*pad)+pad)+'px';
  }

  function reset(){
    ({width:w, height:h} = bounds());
    x = w*0.5; y = h*0.6; vx=vy=0; collected = [];
    targets.forEach(t => { t.classList.remove('caught'); place(t); });
    ship.style.left = x+'px'; ship.style.top = y+'px';
    gotEl.textContent = '—';
    done.hidden = true;
  }

  // simple physics
  const speed = 0.9, friction=0.88, maxv=4.5;
  function step(){
    x += vx; y += vy; vx *= friction; vy *= friction;
    // keep inside
    if(x<20){x=20; vx=0} if(y<20){y=20; vy=0}
    if(x>w-20){x=w-20; vx=0} if(y>h-20){y=h-20; vy=0}
    ship.style.left = x+'px'; ship.style.top = y+'px';
    // collisions
    targets.forEach(t=>{
      if(t.classList.contains('caught')) return;
      const tx = t.offsetLeft+18, ty=t.offsetTop+18;
      if( (x-tx)**2 + (y-ty)**2 < 34**2 ){
        t.classList.add('caught');
        collected.push(t.dataset.v);
        gotEl.textContent = collected.join('-');
        if(collected.slice(0,3).join('')==='404'){
          done.hidden = false;
        }
      }
    });
    requestAnimationFrame(step);
  }

  // keyboard controls
  const keys = {};
  window.addEventListener('keydown', e=>{
    keys[e.key] = true;
    if(['ArrowUp','ArrowDown','ArrowLeft','ArrowRight',' '].includes(e.key)) e.preventDefault();
  }, {passive:false});
  window.addEventListener('keyup', e=> keys[e.key]=false);

  function drive(){
    if(keys.ArrowLeft)  vx = Math.max(-maxv, vx - speed);
    if(keys.ArrowRight) vx = Math.min( maxv, vx + speed);
    if(keys.ArrowUp)    vy = Math.max(-maxv, vy - speed);
    if(keys.ArrowDown)  vy = Math.min( maxv, vy + speed);
    requestAnimationFrame(drive);
  }

  window.addEventListener('resize', reset);
  scene.addEventListener('click', ()=>scene.focus());
  again?.addEventListener('click', reset);

  reset(); step(); drive();
})();
</script>
